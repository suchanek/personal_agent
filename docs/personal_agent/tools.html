248    def _run_async(self, coro):
249        """Helper to run async functions, handling existing event loops."""
250        try:
251            # Try to get the current event loop
252            loop = asyncio.get_running_loop()
253            # If we're in a running loop, we need to use a different approach
254            import concurrent.futures
255            import threading
256
257            # Create a new event loop in a separate thread
258            def run_in_thread():
259                new_loop = asyncio.new_event_loop()
260                asyncio.set_event_loop(new_loop)
261                try:
262                    return new_loop.run_until_complete(coro)
263                finally:
264                    new_loop.close()
265
266            with concurrent.futures.ThreadPoolExecutor() as executor:
267                future = executor.submit(run_in_thread)
268                return future.result()
269
270        except RuntimeError:
271            # No running event loop, safe to use asyncio.run()
272            return asyncio.run(coro)
