 71    def _get_memory_manager_and_db(self):
 72        """Get memory manager and db, ensuring agent is initialized first."""
 73        if not self.agent:
 74            return None, None
 75
 76        # Check if agent is already initialized to avoid redundant initialization
 77        is_initialized = getattr(self.agent, "_initialized", False)
 78
 79        # Only initialize if not already initialized
 80        if not is_initialized and hasattr(self.agent, "_ensure_initialized"):
 81            try:
 82                logger.warning(
 83                    "Agent not initialized, triggering lazy initialization..."
 84                )
 85                asyncio.run(self.agent._ensure_initialized())
 86            except Exception as e:
 87                logger.error(f"Failed to initialize agent: {e}")
 88                return None, None
 89        elif is_initialized:
 90            logger.info("Agent already initialized, skipping initialization")
 91
 92        # Now check for memory components
 93        if hasattr(self.agent, "agno_memory") and self.agent.agno_memory:
 94            return self.agent.agno_memory.memory_manager, self.agent.agno_memory.db
 95        return None, None

323    def _find_tool_by_name(self, tool_name: str):
324        """Find a tool function by name from the agent's tools, similar to CLI approach."""
325        try:
326            # Handle team wrapper case
327            if hasattr(self.agent, "team") and hasattr(self.agent.team, "members"):
328                # Search in the knowledge agent (first team member)
329                if self.agent.team.members:
330                    knowledge_agent = self.agent.team.members[0]
331                    if hasattr(knowledge_agent, "agent") and hasattr(
332                        knowledge_agent.agent, "tools"
333                    ):
334                        for toolkit in knowledge_agent.agent.tools:
335                            # Check if the toolkit itself is a callable tool
336                            if getattr(toolkit, "__name__", "") == tool_name:
337                                return toolkit
338
339                            # Check for tools within the toolkit
340                            if hasattr(toolkit, "tools") and isinstance(
341                                toolkit.tools, list
342                            ):
343                                for tool in toolkit.tools:
344                                    if (
345                                        hasattr(tool, "__name__")
346                                        and tool.__name__ == tool_name
347                                    ):
348                                        return tool
349
350            # Handle direct agent case
351            elif hasattr(self.agent, "agent") and hasattr(self.agent.agent, "tools"):
352                for toolkit in self.agent.agent.tools:
353                    # Check if the toolkit itself is a callable tool
354                    if getattr(toolkit, "__name__", "") == tool_name:
355                        return toolkit
356
357                    # Check for tools within the toolkit
358                    if hasattr(toolkit, "tools") and isinstance(toolkit.tools, list):
359                        for tool in toolkit.tools:
360                            if hasattr(tool, "__name__") and tool.__name__ == tool_name:
361                                return tool
362
363            return None
364        except Exception as e:
365            logger.error(f"Error finding tool {tool_name}: {e}")
366            return None

394        try:

405                return True, result
